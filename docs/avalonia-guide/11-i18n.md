## 11.2 I18nService

```csharp
public sealed class I18nService : INotifyPropertyChanged
{
    public const string CatalogResourceUri = "avares://CodexSwitch/Assets/i18n/languages.json";

    public static I18nService Current { get; } = LoadDefault();

    private readonly Dictionary<string, Dictionary<string, string>> _translations;

    public event EventHandler? LanguageChanged;

    // 从嵌入资源加载
    public static I18nService LoadDefault()
    {
        var catalog = LoadCatalog();
        var resources = catalog.Languages
            .Select(language => LoadLanguageResource(language.Code))
            .ToArray();
        return new I18nService(catalog, resources);
    }

    // 翻译方法，带回退机制
    public string Translate(string key)
    {
        if (string.IsNullOrWhiteSpace(key)) return "";

        if (TryTranslate(CurrentLanguageCode, key, out var current))
            return current;

        return TryTranslate(DefaultLanguageCode, key, out var fallback)
            ? fallback
            : key;
    }

    // 切换语言
    public bool SetLanguage(string? code)
    {
        var resolved = ResolveLanguageCode(code);
        if (string.Equals(CurrentLanguageCode, resolved, StringComparison.OrdinalIgnoreCase))
            return false;

        CurrentLanguageCode = resolved;
        PropertyChanged?.Invoke(this, new PropertyChangedEventArgs("Item[]"));
        PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(nameof(CurrentLanguageCode)));
        LanguageChanged?.Invoke(this, EventArgs.Empty);
        return true;
    }

    public string this[string key] => Translate(key);
}
```

## 11.3 TrExtension - 自定义 MarkupExtension

```csharp
public sealed class TrExtension : MarkupExtension
{
    public TrExtension() { }
    public TrExtension(string key) { Key = key; }

    public string Key { get; set; } = "";

    public override object ProvideValue(IServiceProvider serviceProvider)
    {
        return new TranslationObservable(Key).ToBinding();
    }

    private sealed class TranslationObservable(string key) : IObservable<string>
    {
        public IDisposable Subscribe(IObserver<string> observer)
        {
            observer.OnNext(I18nService.Current.Translate(key));
            EventHandler handler = (_, _) =>
                observer.OnNext(I18nService.Current.Translate(key));
            I18nService.Current.LanguageChanged += handler;
            return new ActionDisposable(() =>
                I18nService.Current.LanguageChanged -= handler);
        }
    }
}
```

## 11.6 深入：全球化 vs 本地化

### 全球化（Globalization）

全球化是使应用程序能够适应不同区域的过程：

- **文化感知**：日期、时间、数字、货币格式
- **排序规则**：不同语言的排序方式
- **文本方向**：LTR（从左到右）和 RTL（从右到左）

### 本地化（Localization）

本地化是为特定区域提供翻译和资源的过程：

- **字符串翻译**：UI 文本的翻译
- **图像本地化**：不同区域使用不同图像
- **布局调整**：RTL 语言需要镜像布局

### .NET 中的文化信息

## 11.7 深入：RTL 支持

### 流向性（FlowDirection）

Avalonia 支持 RTL 布局：

```xml
<!-- 设置整个窗口为 RTL -->
<Window FlowDirection="RightToLeft">
    <StackPanel>
        <TextBlock Text="مرحبا"/>  <!-- 阿拉伯文 -->
        <Button Content="انقر هنا"/>
    </StackPanel>
</Window>
```

### 镜像布局

RTL 布局会自动镜像：

```
LTR 布局：
┌─────────────────────┐
│ [Icon] Title        │
│ [Icon] Description  │
└─────────────────────┘

RTL 布局：
┌─────────────────────┐
│        Title [Icon] │
│  Description [Icon] │
└─────────────────────┘
```

### 处理 RTL 特定样式

```xml
<!-- LTR 特定样式 -->
<Style Selector="TextBlock">
    <Setter Property="Margin" Value="0,0,10,0"/>
</Style>

<!-- RTL 特定样式 -->
<Style Selector="TextBlock">
    <Style.Resources>
        <Thickness x:Key="TextMargin"
                   x:FlowDirection="LeftToRight">0,0,10,0</Thickness>
        <Thickness x:Key="TextMargin"
                   x:FlowDirection="RightToLeft">10,0,0,0</Thickness>
    </Style.Resources>
</Style>
```

## 11.8 深入：复数形式

### 复数规则

不同语言有不同的复数规则：

- **英语**：1 item, 2 items（两种形式）
- **中文**：1 个项目, 2 个项目（一种形式）
- **阿拉伯语**：0 个项目, 1 个项目, 2 个项目, 3-10 个项目, 11-99 个项目, 100+ 个项目（六种形式）

### 实现复数支持

```csharp
public class PluralizationService
{
    private readonly Dictionary<string, Dictionary<PluralForm, string>> _rules;

    public string GetPluralForm(string key, int count)
    {
        var form = GetPluralForm(count);
        if (_rules.TryGetValue(key, out var forms) &&
            forms.TryGetValue(form, out var text))
        {
            return text.Replace("{count}", count.ToString());
        }
        return key;
    }

    private PluralForm GetPluralForm(int count)
    {
        // 简化的英语规则
        return count switch
        {
            0 => PluralForm.Zero,
            1 => PluralForm.One,
            _ => PluralForm.Other
        };
    }
}

public enum PluralForm
{
    Zero,
    One,
    Two,
    Few,
    Many,
    Other
}
```

### 在 AXAML 中使用复数

```xml
<!-- 定义复数形式 -->
<Style.Resources>

    <x:String x:Key="items.one">1 个项目</x:String>
    <x:String x:Key="items.other">{count} 个项目</x:String>
</Style.Resources>
```

## 11.9 深入：日期和数字格式化

### 日期格式化

```csharp
public class DateFormatService
{
    public string FormatDate(DateTime date, string format = "short")
    {
        var culture = CultureInfo.CurrentCulture;
        return format switch
        {
            "short" => date.ToString("d", culture),      // 2025/5/27
            "long" => date.ToString("D", culture),        // 2025年5月27日
            "time" => date.ToString("t", culture),        // 14:30
            "datetime" => date.ToString("f", culture),    // 2025年5月27日 14:30
            "relative" => GetRelativeTime(date),
            _ => date.ToString(format, culture)
        };
    }

    private string GetRelativeTime(DateTime date)
    {
        var span = DateTime.Now - date;
        if (span.TotalMinutes < 1) return "刚刚";
        if (span.TotalMinutes < 60) return $"{(int)span.TotalMinutes} 分钟前";
        if (span.TotalHours < 24) return $"{(int)span.TotalHours} 小时前";
        if (span.TotalDays < 7) return $"{(int)span.TotalDays} 天前";
        return FormatDate(date, "short");
    }
}
```

### 数字格式化

```csharp
public class NumberFormatService
{
    public string FormatNumber(double number, string format = "default")
    {
        var culture = CultureInfo.CurrentCulture;
        return format switch
        {
            "default" => number.ToString("N", culture),     // 1,234,567.89
            "integer" => number.ToString("N0", culture),    // 1,234,568
            "currency" => number.ToString("C", culture),    // ¥1,234.56
            "percent" => number.ToString("P", culture),     // 12.34%
            "compact" => FormatCompact(number),
            _ => number.ToString(format, culture)
        };
    }

    private string FormatCompact(double number)
    {
        if (number >= 1_000_000)
            return $"{number / 1_000_000:F1}M";
        if (number >= 1_000)
            return $"{number / 1_000:F1}K";
        return number.ToString("F0");
    }
}
```

```csharp
public class TranslationValidator
{
    public ValidationResult Validate(string baseLanguage, string targetLanguage)
    {
        var baseKeys = LoadKeys(baseLanguage);
        var targetKeys = LoadKeys(targetLanguage);

        var missingKeys = baseKeys.Except(targetKeys).ToList();
        var extraKeys = targetKeys.Except(baseKeys).ToList();

        return new ValidationResult
        {
            MissingKeys = missingKeys,
            ExtraKeys = extraKeys,
            IsValid = missingKeys.Count == 0
        };
    }
}
```

### 翻译记忆库

```csharp
public class TranslationMemory
{
    private readonly Dictionary<string, Dictionary<string, string>> _memory;

    public string? FindTranslation(string sourceText, string sourceLang, string targetLang)
    {
        if (_memory.TryGetValue(sourceLang, out var translations) &&
            translations.TryGetValue(sourceText, out var targetText))
        {
            return targetText;
        }
        return null;
    }

    public void AddTranslation(string sourceText, string sourceLang,
                               string targetText, string targetLang)
    {
        if (!_memory.ContainsKey(sourceLang))
            _memory[sourceLang] = new Dictionary<string, string>();
        _memory[sourceLang][sourceText] = targetText;
    }
}
```

## 11.12 深入：ICU 消息格式

### ICU MessageFormat

ICU（International Components for Unicode）消息格式是处理复数、性别等复杂情况的标准：

```
{count, plural,
    =0 {没有项目}
    one {1 个项目}
    other {{count} 个项目}
}
```

### 实现 ICU 支持

```csharp
public class IcuMessageFormatter
{
    public string Format(string pattern, Dictionary<string, object> args)
    {
        var result = pattern;

        // 处理简单替换
        foreach (var (key, value) in args)
        {
            result = result.Replace($"{{{key}}}", value.ToString());
        }

        // 处理复数
        result = ProcessPlural(result, args);

        return result;
    }

    private string ProcessPlural(string pattern, Dictionary<string, object> args)
    {
        var regex = new Regex(@"\{(\w+), plural, (.+?)\}");
        return regex.Replace(pattern, match =>
        {
            var key = match.Groups[1].Value;
            var pluralForms = ParsePluralForms(match.Groups[2].Value);
            var count = Convert.ToInt32(args[key]);
            var form = GetPluralForm(count);
            return pluralForms.TryGetValue(form, out var text)
                ? text.Replace("{count}", count.ToString())
                : count.ToString();
        });
    }
}
```

### 使用 ICU 格式

```json
  "greeting": "{gender, select, male {他} female {她} other {他们}}好"
```
